# 养老机构预收费资金监管平台 - 开发列表

## 🎯 开发策略说明

### 核心原则
- **单模块开发**: 所有后端代码统一放在 `ruoyi-admin` 模块下，不新建Maven模块
- **功能导向**: ��实现功能为目标，避免过度抽象和设计
- **简洁优先**: 代码保持简洁明了，注重可读性
- **快速迭代**: 实现一个功能，测试一个功能，快速验证

### 项目结构
```
D:\newhm\newzijin\
├── ruoyi-admin\              # 后端主模块（所有后端代码放这里）
```

### 开发方式
- 按功能模块逐步开发，每个功能完成后立即测试
- 前后端配合开发，先实现核心功能再优化体验
- 复用若依现有框架和组件，减少重复开发

```sql
CREATE TABLE `pension_institution` (
  `institution_id` bigint(20) NOT NULL AUTO_INCREMENT COMMENT '机构ID',
  `institution_name` varchar(200) NOT NULL COMMENT '机构名称',
  `credit_code` varchar(50) NOT NULL COMMENT '统一信用代码',
  `registered_capital` decimal(15,2) DEFAULT NULL COMMENT '注册资金(万元)',
  `registered_address` varchar(500) DEFAULT NULL COMMENT '注册地址',
  `contact_person` varchar(50) DEFAULT NULL COMMENT '联系人',
  `contact_phone` varchar(20) DEFAULT NULL COMMENT '联系电话',
  `business_scope` varchar(1000) DEFAULT NULL COMMENT '经营范围',
  `bed_count` int(11) DEFAULT NULL COMMENT '床位数',
  `fixed_assets` decimal(15,2) DEFAULT NULL COMMENT '固定资产净额(万元)',
  `status` char(1) DEFAULT '0' COMMENT '状态(0待审批 1已入驻 2已驳回)',
  `apply_time` datetime DEFAULT NULL COMMENT '申请时间',
  `approve_time` datetime DEFAULT NULL COMMENT '审批时间',
  `approve_user` varchar(64) DEFAULT NULL COMMENT '审批人',
  `approve_remark` varchar(500) DEFAULT NULL COMMENT '审批意见',
  `create_by` varchar(64) DEFAULT '' COMMENT '创建者',
  `create_time` datetime DEFAULT NULL COMMENT '创建时间',
  `update_by` varchar(64) DEFAULT '' COMMENT '更新者',
  `update_time` datetime DEFAULT NULL COMMENT '更新时间',
  `remark` varchar(500) DEFAULT NULL COMMENT '备注',
  PRIMARY KEY (`institution_id`),
  UNIQUE KEY `uk_credit_code` (`credit_code`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='养老机构信息表';

-- 机构附件材料表
CREATE TABLE `pension_institution_attach` (
  `attach_id` bigint(20) NOT NULL AUTO_INCREMENT COMMENT '附件ID',
  `institution_id` bigint(20) NOT NULL COMMENT '机构ID',
  `attach_type` char(1) DEFAULT NULL COMMENT '附件类型(1营业执照 2批准证书 3三方协议)',
  `attach_name` varchar(200) DEFAULT NULL COMMENT '附件名称',
  `attach_path` varchar(500) DEFAULT NULL COMMENT '附件路径',
  `file_size` bigint(20) DEFAULT NULL COMMENT '文件大小(字节)',
  `create_by` varchar(64) DEFAULT '' COMMENT '上传者',
  `create_time` datetime DEFAULT NULL COMMENT '上传时间',
  PRIMARY KEY (`attach_id`),
  KEY `idx_institution_id` (`institution_id`)
```

```java
import com.ruoyi.common.utils.poi.ExcelUtil;
import com.ruoyi.common.core.page.TableDataInfo;

@Controller
@RequestMapping("/pension/institution")
public class PensionInstitutionController extends BaseController {
    private String prefix = "pension/institution";

    @Autowired
    private IPensionInstitutionService pensionInstitutionService;

    @RequiresPermissions("pension:institution:view")
    @GetMapping()
    public String institution() {
        return prefix + "/institution";
    }

    @RequiresPermissions("pension:institution:list")
    @PostMapping("/list")
    @ResponseBody
    public TableDataInfo list(PensionInstitution pensionInstitution) {
        startPage();
        List<PensionInstitution> list = pensionInstitutionService.selectPensionInstitutionList(pensionInstitution);
        return getDataTable(list);
    }

    @GetMapping("/add")
    public String add() {
        return prefix + "/add";
    }

    @RequiresPermissions("pension:institution:add")
    @Log(title = "养老机构入驻申请", businessType = BusinessType.INSERT)
    @PostMapping("/add")
    @ResponseBody
    public AjaxResult addSave(PensionInstitution pensionInstitution) {
        return toAjax(pensionInstitutionService.insertPensionInstitution(pensionInstitution));
    }

    @GetMapping("/edit/{institutionId}")
    public String edit(@PathVariable("institutionId") Long institutionId, ModelMap mmap) {
        PensionInstitution pensionInstitution = pensionInstitutionService.selectPensionInstitutionByInstitutionId(institutionId);
        mmap.put("pensionInstitution", pensionInstitution);
        return prefix + "/edit";
    }

    @RequiresPermissions("pension:institution:edit")
    @Log(title = "修改养老机构", businessType = BusinessType.UPDATE)
    @PostMapping("/edit")
    @ResponseBody
    public AjaxResult editSave(PensionInstitution pensionInstitution) {
        return toAjax(pensionInstitutionService.updatePensionInstitution(pensionInstitution));
    }

    @RequiresPermissions("pension:institution:remove")
    @Log(title = "删除养老机构", businessType = BusinessType.DELETE)
    @PostMapping("/remove")
    @ResponseBody
    public AjaxResult remove(String ids) {
        return toAjax(pensionInstitutionService.deletePensionInstitutionByInstitutionIds(ids));
    }
}
```

```java
// PensionInstitutionServiceImpl.java
package com.ruoyi.pension.service.impl;

import java.util.List;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import com.ruoyi.pension.mapper.PensionInstitutionMapper;
import com.ruoyi.pension.domain.PensionInstitution;
import com.ruoyi.pension.service.IPensionInstitutionService;

@Service
public class PensionInstitutionServiceImpl implements IPensionInstitutionService {
    @Autowired
    private PensionInstitutionMapper pensionInstitutionMapper;

    @Override
    public List<PensionInstitution> selectPensionInstitutionList(PensionInstitution pensionInstitution) {
        return pensionInstitutionMapper.selectPensionInstitutionList(pensionInstitution);
    }

    @Override
    public PensionInstitution selectPensionInstitutionByInstitutionId(Long institutionId) {
        return pensionInstitutionMapper.selectPensionInstitutionByInstitutionId(institutionId);
    }

    @Override
    public int insertPensionInstitution(PensionInstitution pensionInstitution) {
        pensionInstitution.setStatus("0"); // 设置为待审批状态
        pensionInstitution.setApplyTime(new java.util.Date());
        return pensionInstitutionMapper.insertPensionInstitution(pensionInstitution);
    }

    @Override
    public int updatePensionInstitution(PensionInstitution pensionInstitution) {
        return pensionInstitutionMapper.updatePensionInstitution(pensionInstitution);
    }

    @Override
    public int deletePensionInstitutionByInstitutionIds(String ids) {
        return pensionInstitutionMapper.deletePensionInstitutionByInstitutionIds(Convert.toStrArray(ids));
    }
}
```

```vue
<script>
import { listInstitution, getInstitution, delInstitution, addInstitution, updateInstitution } from "@/api/pension/institution";

export default {
  name: "Institution",
  dicts: ['sys_notice_status'],
  data() {
    return {
      loading: true,
      ids: [],
      showSearch: true,
      total: 0,
      institutionList: [],
      title: "",
      open: false,
      queryParams: {
        pageNum: 1,
        pageSize: 10,
        institutionName: null,
        creditCode: null,
        status: null
      },
      form: {},
      rules: {
        institutionName: [
          { required: true, message: "机构名称不能为空", trigger: "blur" }
        ],
        creditCode: [
          { required: true, message: "统一信用代码不能为空", trigger: "blur" }
        ],
        contactPerson: [
          { required: true, message: "联系人不能为空", trigger: "blur" }
        ],
        contactPhone: [
          { required: true, message: "联系电话不能为空", trigger: "blur" }
        ]
      }
    };
  },
  created() {
    this.getList();
  },
  methods: {
    getList() {
      this.loading = true;
      listInstitution(this.queryParams).then(response => {
        this.institutionList = response.rows;
        this.total = response.total;
        this.loading = false;
      });
    },
    cancel() {
      this.open = false;
      this.reset();
    },
    reset() {
      this.form = {
        institutionId: null,
        institutionName: null,
        creditCode: null,
        registeredCapital: null,
        registeredAddress: null,
        contactPerson: null,
        contactPhone: null,
        bedCount: null,
        fixedAssets: null,
        status: null,
        applyTime: null,
        approveTime: null,
        approveUser: null,
        approveRemark: null,
        createBy: null,
        createTime: null,
        updateBy: null,
        updateTime: null,
        remark: null
      };
      this.resetForm("form");
    },
    handleQuery() {
      this.queryParams.pageNum = 1;
      this.getList();
    },
    resetQuery() {
      this.resetForm("queryForm");
      this.handleQuery();
    },
    handleAdd() {
      this.reset();
      this.open = true;
      this.title = "添加养老机构入驻申请";
    },
    handleUpdate(row) {
      this.reset();
      const institutionId = row.institutionId || this.ids
      getInstitution(institutionId).then(response => {
        this.form = response.data;
        this.open = true;
        this.title = "修改养老机构";
      });
    },
    submitForm() {
      this.$refs["form"].validate(valid => {
        if (valid) {
          if (this.form.institutionId != null) {
            updateInstitution(this.form).then(response => {
              this.$modal.msgSuccess("修改成功");
              this.open = false;
              this.getList();
            });
          } else {
            addInstitution(this.form).then(response => {
              this.$modal.msgSuccess("新增成功");
              this.open = false;
              this.getList();
            });
          }
        }
      });
    },
    handleDelete(row) {
      const institutionIds = row.institutionId || this.ids;
      this.$modal.confirm('是否确认删除养老机构编号为"' + institutionIds + '"的数据项？').then(function() {
        return delInstitution(institutionIds);
      }).then(() => {
        this.getList();
        this.$modal.msgSuccess("删除成功");
      }).catch(() => {});
    }
  }
};
</script>
```
